function doPost(e) {
  try {
    let data;
    
    // Method 1: Parse from postData.contents (XHR/Fetch)
    if (e.postData && e.postData.contents) {
      data = JSON.parse(e.postData.contents);
    }
    // Method 2: Parse from parameter (Form submission)
    else if (e.parameter.data) {
      data = JSON.parse(e.parameter.data);
    }
    // Method 3: Parse from form plain text
    else if (e.parameter.json) {
      data = JSON.parse(e.parameter.json);
    }
    // Method 4: Direct parameters (fallback)
    else {
      data = e.parameter;
    }
    
    // Validasi
    if (!data.filename || !data.mimeType || !data.bytes) {
      return jsonResponse({ status: "error", message: "Missing required fields." });
    }
    
    // Upload ke Google Drive
    const folder = DriveApp.getFolderById("1tqTdK1ReLTZ_geFpby8z5dGIN2qeAZFm");
    const decoded = Utilities.base64Decode(data.bytes);
    const blob = Utilities.newBlob(decoded, data.mimeType, data.filename);
    const file = folder.createFile(blob);
    
    return jsonResponse({
      status: "success",
      message: "File uploaded successfully.",
      fileName: file.getName(),
      fileUrl: file.getUrl()
    });
  } catch (error) {
    return jsonResponse({ status: "error", message: error.toString() });
  }
}

function jsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
